import math

# Константы
SERVO_CHANNEL = 2
SENSOR_CHANNEL = 1
INITIAL_ANGLE = 0
ANGLE_STEP = 15
MIN_ANGLE = -180
MAX_ANGLE = 0
MOTOR_POWER_FORWARD = 100
MOTOR_POWER_BACKWARD = -100
SPEED_MULTIPLIER = 0.7
ANGLE_THRESHOLD = -90
MM_TO_M = 1000

# Переменные
global current_angle, direction, step, is_scanning, found_cube
current_angle = INITIAL_ANGLE
direction = -1
step = ANGLE_STEP
is_scanning = True
found_cube = False

def move_to_cube(distance, angle):
    if 90 > angle < 270:
        robot.set_motor_power(MOTOR_POWER_BACKWARD, distance // MM_TO_M)
    else:
        robot.set_motor_power(MOTOR_POWER_FORWARD, distance // MM_TO_M)
    robot.stop()
    print(f"Куб обнаружен на {angle}°; дистанция {distance} мм; подъезд завершён.")


def handle_servo():
    global current_angle, direction, step, is_scanning, found_cube

    robot.set_servo_angle(2, current_angle)
    d = robot.get_sensor(1)
    if d > 0:
        move_to_cube(d, current_angle)
        return
    else:
        current_angle += ANGLE_STEP

        if current_angle > 360:
            current_angle = MAX_ANGLE

# Функция on_servo_rotate устаналивает циклический вызов функции handle_servo с интервалом 500мс.
# Это является аналогом следующего цикла: while True: handle_servo() time.sleep(0.5)
# Необходимо для корректного получения данных от датчиков в условиях данного симулятора.
on_servo_rotate(handle_servo)

    dist_mm = robot.get_sensor(1)

def drive_to_star(x_star, z_star):
    pos = robot.get_gps()
    x0, y0, z0 = pos['lon'], pos['alt'], pos['lat']

    dx = x_star - x0
    dz = z_star - z0
    target_angle = math.degrees(math.atan2(dx, dz))
    heading = robot.get_gyro_yaw()
    turn_angle = target_angle - heading

    if (turn_angle < -180):
        turn_angle = turn_angle + 360
    if (turn_angle > 180):
        turn_angle = turn_angle - 360

    finished_angle = turn_angle + 90
    robot.set_servo_angle(1, finished_angle)
    robot.set_motor_power(100, 0.1)
    return

        x_s, z_s = star['x'], star['z']
        drive_to_star(x_s, z_s)
    else:
        robot.stop()

    """Вычисляет стоимость доставки для указанного количества товаров"""

    elif count == 1:
