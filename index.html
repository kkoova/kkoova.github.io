def find_shortest_path(grid):
    rows = len(grid)
    cols = len(grid[0])
    
    # 1. Найти координаты start и goal
    start = goal = None
    for r in range(rows):
        for c in range(cols):
            if grid[r][c] == 'S':
                start = (r, c)
            elif grid[r][c] == 'F':
                goal = (r, c)
    
    if not start or not goal:
        return []
    
    # Направления: вверх, вниз, влево, вправо
    directions = [(-1, 0), (1, 0), (0, -1), (0, 1)]
    
    # 2. BFS с очередью и словарем для восстановления пути
    queue = deque([start])
    came_from = {start: None}  # Словарь родительских координат
    
    while queue:
        current = queue.popleft()
        
        if current == goal:
            break
            
        r, c = current
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            
            # Проверка границ и проходимости
            if (0 <= nr < rows and 0 <= nc < cols and 
                grid[nr][nc] != '#' and (nr, nc) not in came_from):
                
                queue.append((nr, nc))
                came_from[(nr, nc)] = current

        return []  # Путь не найден
    
    path = []

    while current is not None:
        path.append(current)
        current = came_from[current]
    
    return path[::-1]  # Перевернуть, чтобы от старта к финишу

    global is_moving
  
    if not target_list:
        return

    target = min(
        target_list,
        key=lambda c: math.hypot(
            c['x'] - robot.getPositionX(),
            c['z'] - robot.getPositionZ()
        )
    )
  
    # Вычисляем расстояние до выбранной цели
    dist = math.hypot(
        target['x'] - robot.getPositionX(),
        target['z'] - robot.getPositionZ()
    )

    if dist < TARGET_DISTANCE:
      is_moving = False
      robot.stop()

      # Если робот собирает груз, считываем данные и сохраняем их

SENSOR_HISTORY = 3
sensor_history = []
SENSOR_WEIGHTS = [-3, -1, 1, 3]
LOST_THRESHOLD = 50
I_MAX = 100

# Настройки скорости
BASE_SPEED = 50
MIN_SPEED = 40
MAX_SPEED = 60
K_SPEED = 1

global integral, prev_error, last_valid
integral = 0
prev_error = 0
last_valid = 0

def clamp(value, min_val, max_val):
    return max(min_val, min(max_val, value))

def filtered_sensors(raw_sensors):
    sensor_history.append(raw_sensors)
    if len(sensor_history) > SENSOR_HISTORY:
        sensor_history.pop(0)
    return [sum(vals)/len(vals) for vals in zip(*sensor_history)]

def calculate_position(sensors):
    inverted = [255 - s for s in sensors]
    total = sum(inverted)
    if total == 0:
        return 0
    weighted_sum = sum(w * val for w, val in zip(SENSOR_WEIGHTS, inverted))
    return weighted_sum / total

def enhanced_pid(raw_sensors):
    global integral, prev_error, last_valid

    filtered = filtered_sensors(raw_sensors)
    position = calculate_position(filtered)

    if sum(filtered) < LOST_THRESHOLD:
        error = last_valid * 1.5
    else:
        error = position
        last_valid = position

    # PID
    P = KP * error
    integral = clamp(integral + error, -I_MAX, I_MAX)
    I = KI * integral
    D = KD * (error - prev_error)
    prev_error = error

    correction = P + I + D

    speed = clamp(BASE_SPEED - abs(error) * K_SPEED, MIN_SPEED, MAX_SPEED)
    servo_angle = clamp(90 - correction, 0, 180)
    return servo_angle, speed

    servo_angle, speed = enhanced_pid(sensors_values)
    robot.set_motor_power(int(speed))
    robot.set_servo_angle(1, int(servo_angle))
