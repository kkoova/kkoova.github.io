«Робот учится видеть и подъезжать к цели»

Код для симулятора:
Для выполнение данного симулятора вам обязательно надо будет заехать в красный квадрат до того момента как он станет зеленым. После, подожать выполнения цикла и нажать: "ESC". Выполнение симулятора должно быть выполнено на 80 - 90%
import math
import time

CHECK_INTERVAL = 2000
measurements = {}

def handle_servo_rotate(angle):
    print(f"{angle}° → {robot.get_sensor(1)} мм")
    measurements[angle] = robot.get_sensor(1)

i = 0

while i < 361:
    robot.set_servo_angle(2, i) 
    handle_servo_rotate(i) 
    i += 60
    time.sleep(CHECK_INTERVAL/1000)

nearest_angle, nearest_dist = min(
    measurements.items(),
    key=lambda item: item[1] 
)

print(f"Ближайшее препятствие под {nearest_angle}° на {nearest_dist} мм")

import time 
CHECK_INTERVAL = 5000

def handle_battery_charge():
    battery = robot.get_sensor(3)
    print(f"\nТекущий заряд: {battery}%")

    if battery >= 90:
        print("Аккумулятор полностью заряжен.")
    elif battery >= 50:
        print("Аккумулятор заряжен более чем наполовину.")
    elif battery >= 10:
        print("Аккумулятор разряжен, скоро нужна подзарядка.")
    else:
        print("Критический уровень заряда! Срочно заряжаемся.")

on_battery_charge_changed(handle_battery_charge)
