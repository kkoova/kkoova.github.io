Код для симулятора:
Для выполнение данного симулятора вам обязательно надо будет заехать в красный квадрат до того момента как он станет зеленым.
import math

# Константы
SERVO_CHANNEL = 2
SENSOR_CHANNEL = 1
INITIAL_ANGLE = 0
ANGLE_STEP = 15
MIN_ANGLE = -180
MAX_ANGLE = 0
MOTOR_POWER_FORWARD = 100
MOTOR_POWER_BACKWARD = -100
SPEED_MULTIPLIER = 0.7
ANGLE_THRESHOLD = -90
MM_TO_M = 1000

# Переменные
global current_angle, direction, step, is_scanning, found_cube
current_angle = INITIAL_ANGLE
direction = -1
step = ANGLE_STEP
is_scanning = True
found_cube = False

def move_to_cube(distance, angle):
    if 90 > angle < 270:
        robot.set_motor_power(MOTOR_POWER_BACKWARD, distance // MM_TO_M)
    else:
        robot.set_motor_power(MOTOR_POWER_FORWARD, distance // MM_TO_M)
    robot.stop()
    print(f"Куб обнаружен на {angle}°; дистанция {distance} мм; подъезд завершён.")


def handle_servo():
    global current_angle, direction, step, is_scanning, found_cube

    robot.set_servo_angle(2, current_angle)
    d = robot.get_sensor(1)
    if d > 0:
        move_to_cube(d, current_angle)
        return
    else:
        current_angle += ANGLE_STEP

        if current_angle > 360:
            current_angle = MAX_ANGLE

# Функция on_servo_rotate устаналивает циклический вызов функции handle_servo с интервалом 500мс.
# Это является аналогом следующего цикла: while True: handle_servo() time.sleep(0.5)
# Необходимо для корректного получения данных от датчиков в условиях данного симулятора.
on_servo_rotate(handle_servo)

def handle_servo_rotate():
    cur_angle = robot.get_servo_angle(2)
    measurements[cur_angle] = robot.get_sensor(1)

Код для симулятора:
Для выполнение данного симулятора вам обязательно доехать до финиша (WASD), следуя указателю и сделать 3 коппии карты (3 раза запустить код)
import math

# Переменные для отрисовки карты
SIZE = 41
CENTER = SIZE // 2
SCALE = 0.1

field = [[" "]*SIZE for _ in range(SIZE)]   # чистое 41×41 поле
field[CENTER][CENTER] = "R"

#field = [['ㅤ']*SIZE for _ in range(SIZE)]

robot.set_servo_angle(2, 0)

def print_map():
    field[CENTER][CENTER] = 'R'

    for row in field:
        print("".join(row))

    robot.set_servo_angle(2, 0)
    stop_handler_frame()

def handle_frame():
    global field, SIZE, CENTER, SCALE
    current_angle = robot.get_servo_angle(2)

    if (current_angle >= 360):
      print_map()
      return
    
    robot.set_servo_angle(2, current_angle)
    dist_mm = robot.get_sensor(1)

    if dist_mm > 0:
        dist_m = dist_mm / 1000

        rad = math.radians(current_angle)
        x = dist_m * math.cos(rad)
        y = dist_m * math.sin(rad)

        xi = int(round(CENTER + x / SCALE))
        yi = int(round(CENTER - y / SCALE))

        if 0 <= xi < SIZE and 0 <= yi < SIZE:
            field[yi][xi] = "*"
    
    robot.set_servo_angle(2, current_angle + 10)

# Функция on_frame устаналивает циклический вызов функции handle_frame с интервалом 200мс.
# Это является аналогом следующего цикла: while True: handle_frame()
# Необходимо для корректного получения данных от датчиков в условиях данного симулятора.
on_frame(handle_frame)
